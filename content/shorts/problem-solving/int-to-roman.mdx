  function intToRomanA(num) {
    const values = [1000,900,500,400,100,90,50,40,10,9,5,4,1];
    const symbols = ["M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"];
    let res = "";
    for (let i=0;i<values.length && num>0;i++){
        while(num>=values[i]){ res += symbols[i]; num -= values[i]; }
    }
    console.log(res);
  }

  // Alternative approach using a map-like structure
  function intToRomanB(num) { 
    const map = [[1000,"M"],[900,"CM"],[500,"D"],[400,"CD"],[100,"C"],[90,"XC"],
                [50,"L"],[40,"XL"],[10,"X"],[9,"IX"],[5,"V"],[4,"IV"],[1,"I"]];
    let res = "";
    for (const [v,s] of map){
      while (num >= v) { res += s; num -= v; }
    }
    return res;
  }

  // Recursive approach
  function intToRomanC(num) {
    const values = [1000,900,500,400,100,90,50,40,10,9,5,4,1];
    const symbols = ["M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"];
    function helper(n, idx) {
      if (n===0) return "";
      while (idx<values.length && n<values[idx]) idx++;
      return symbols[idx] + helper(n-values[idx], idx);
    }
    return helper(num,0);
  }

  // Digit mapping approach
  function intToRomanD(num) {
    const TH = ["","M","MM","MMM"];
    const H  = ["","C","CC","CCC","CD","D","DC","DCC","DCCC","CM"];
    const T  = ["","X","XX","XXX","XL","L","LX","LXX","LXXX","XC"];
    const O  = ["","I","II","III","IV","V","VI","VII","VIII","IX"];
    return TH[Math.floor(num/1000)] + H[Math.floor((num%1000)/100)] +
          T[Math.floor((num%100)/10)] + O[num%10];
  }

  // Example
  intToRomanA(1994); // "MCMXCIV"

  public int romanToInt(String s) {
      Map<Character, Integer> mp = Map.of(
          'I', 1, 'V', 5, 'X', 10, 'L', 50,
          'C', 100, 'D', 500, 'M', 1000
      );
      int total = 0;
      for (int i = 0; i < s.length(); i++) {
          int curr = mp.get(s.charAt(i));
          int next = (i + 1 < s.length()) ? mp.get(s.charAt(i + 1)) : 0;
          if (curr < next) total -= curr;
          else total += curr;
      }
      return total;
  }
